// Whitepaper-compliant vesting calculation
function _computeWhitepaperReleasableAmount(
VestingSchedule memory schedule
) internal view returns (uint256) {
uint256 timeElapsed = block.timestamp - schedule.startTime;
// Calculate completed 6-month intervals
uint256 intervalsPassed = timeElapsed / UNLOCK_INTERVAL;
// Handle completion (5 years passed)
if (timeElapsed >= VESTING_DURATION) {
return schedule.totalAmount - schedule.releasedAmount;
}
// Calculate total unlocked: 2.5% per interval
uint256 totalUnlockedPercentage = intervalsPassed * UNLOCK_PERCENTAGE_PER_INTERVAL;
// Cap at 100% (10,000 basis points)
if (totalUnlockedPercentage > BASIS_POINTS) {
totalUnlockedPercentage = BASIS_POINTS;
}
uint256 totalUnlockedAmount =
(schedule.totalAmount * totalUnlockedPercentage) / BASIS_POINTS;
return totalUnlockedAmount - schedule.releasedAmount;
}
Mathematical Precision: Our vesting algorithm uses basis points (10,000 = 100%) to ensure exact 2.5% releases without rounding errors, maintaining perfect mathematical accuracy over the 5-year period.

function _getETHPriceInUSD() internal view returns (uint256) {

uint80 answeredInRound
